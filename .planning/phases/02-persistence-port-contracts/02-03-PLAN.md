```typescript
import * as fs from 'node:fs/promises';
import * as path from 'node:path';
import { GvcConfigSchema, type GvcConfig } from './schema.js';

export const DEFAULT_CONFIG_PATH = 'gvc0.config.json';
const LEGACY_CONFIG_PATH = '.gvc0/config.json';  // transitional — old src/config.ts default

export interface ConfigLoader {
  load(): Promise<GvcConfig>;
}

export interface ConfigSource extends ConfigLoader {
  watch(): { close(): void };
}

export class JsonConfigLoader implements ConfigSource {
  constructor(private readonly configPath: string = DEFAULT_CONFIG_PATH) {}

  async load(): Promise<GvcConfig> {
    const resolved = path.resolve(this.configPath);
    let raw: string;
    try {
      raw = await fs.readFile(resolved, 'utf-8');
    } catch (err) {
      if (isEnoent(err)) {
        // Fall back to legacy path if primary missing.
        const legacy = path.resolve(LEGACY_CONFIG_PATH);
        try {
          raw = await fs.readFile(legacy, 'utf-8');
        } catch {
          throw new Error(
            `Config file not found at ${resolved} (or legacy ${legacy}). ` +
              `Create gvc0.config.json with at minimum a \`models\` map for all four roles.`,
          );
        }
      } else {
        throw err;
      }
    }
    let parsed: unknown;
    try {
      parsed = JSON.parse(raw);
    } catch (err) {
      throw new Error(`Invalid JSON in ${resolved}: ${(err as Error).message}`);
    }
    const result = GvcConfigSchema.safeParse(parsed);
    if (!result.success) {
      const firstIssue = result.error.issues[0];
      const fieldPath = firstIssue.path.join('.');
      throw new Error(
        `Invalid config at ${resolved}: field \`${fieldPath}\` — ${firstIssue.message}`,
      );
    }
    return result.data;
  }

  watch(): { close(): void } {
    // Phase-2 stub — see RESEARCH Open Question #3 + CONTEXT (F).
    // Phase 7 will replace this with real fs.watch wiring.
    return { close() {} };
  }
}

function isEnoent(err: unknown): boolean {
  return typeof err === 'object' && err !== null && (err as NodeJS.ErrnoException).code === 'ENOENT';
}
```

const validMinimal = {

import { mkdtempSync, rmSync, writeFileSync } from 'node:fs';
import { tmpdir } from 'node:os';
import { join } from 'node:path';
import { describe, it, expect } from 'vitest';
import { JsonConfigLoader } from '@config/load';

describe('config boot integration', () => {
  it('loads + validates a user-authored gvc0.config.json', async () => {
    const dir = mkdtempSync(join(tmpdir(), 'gvc0-config-'));
    try {
      const configPath = join(dir, 'gvc0.config.json');
      writeFileSync(configPath, JSON.stringify({
        dbPath: './custom/state.db',
        models: {
          topPlanner: { provider: 'anthropic', model: 'sonnet' },
          featurePlanner: { provider: 'anthropic', model: 'sonnet' },
          taskWorker: { provider: 'anthropic', model: 'haiku' },
          verifier: { provider: 'anthropic', model: 'haiku' },
        },
        workerCap: 8,
        budget: { globalUsd: 50, perTaskUsd: 2 },
      }, null, 2));

      const cfg = await new JsonConfigLoader(configPath).load();
      expect(cfg.dbPath).toBe('./custom/state.db');
      expect(cfg.workerCap).toBe(8);
      expect(cfg.retryCap).toBe(5);  // default
      expect(cfg.models.taskWorker.model).toBe('haiku');
      expect(cfg.budget?.warnAtPercent).toBe(80);  // default
    } finally {
      rmSync(dir, { recursive: true, force: true });

      }));
      await expect(new JsonConfigLoader(configPath).load()).rejects.toThrow(/models|role/i);
    } finally {
      rmSync(dir, { recursive: true, force: true });
    }
  });
});
