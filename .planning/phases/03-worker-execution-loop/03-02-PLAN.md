Create `src/runtime/ipc/quarantine.ts`:

```typescript
import type { Store } from '@orchestrator/ports';

export type Direction = 'parent_from_child' | 'child_from_parent';

export interface QuarantineEntry {
  ts: number;
  direction: Direction;
  agentRunId?: string;
  raw: string;
  errorMessage: string;
}

export interface Quarantine {
  record(entry: QuarantineEntry): void;
  recent(): QuarantineEntry[];
}

export function createQuarantine(opts: {
  store?: Store;
  ringCapacity?: number;
} = {}): Quarantine {
  const capacity = opts.ringCapacity ?? 64;
  const ring: QuarantineEntry[] = [];
  return {
    record(entry) {
      ring.push(entry);
      if (ring.length > capacity) ring.shift();
      // Fire-and-forget async persistence. If store throws, swallow — the
      // ring is the authoritative debug surface per RESEARCH Q#3.
      if (opts.store) {
        queueMicrotask(() => {
          try { opts.store!.appendQuarantinedFrame(entry); }
          catch { /* fire-and-forget */ }
        });
      }
    },
    recent() {
      return [...ring];
    },
  };
}
```

Design notes to encode as comments:
- Ring is authoritative for debugging (in-process visibility).
- Store write is `queueMicrotask` to avoid blocking the readline handler. Swallowed errors are ACCEPTABLE here because the ring still has the data and the Store is a secondary surface.
- `ringCapacity=64` per RESEARCH §NDJSON IPC Framing.
