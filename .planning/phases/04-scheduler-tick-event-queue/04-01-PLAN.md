import { readdirSync, readFileSync, statSync } from 'node:fs';
import { join, relative, sep } from 'node:path';
import * as ts from 'typescript';
import { describe, expect, it } from 'vitest';

import allowlist from './scheduler-boundary-allowlist.json' with { type: 'json' };

const MUTATION_METHODS = new Set([
  'createMilestone', 'createFeature', 'createTask',
  'addDependency', 'removeDependency',
  'splitFeature', 'mergeFeatures',
  'cancelFeature', 'removeFeature',
  'changeMilestone', 'editFeature',
  'addTask', 'editTask', 'removeTask',
  'reorderTasks', 'reweight',
  'queueMilestone', 'dequeueMilestone', 'clearQueuedMilestones',
  'transitionFeature', 'transitionTask',
  'updateMergeTrainState', 'replaceUsageRollups',
]);

function walkTsFiles(dir: string, out: string[] = []): string[] {
  for (const entry of readdirSync(dir)) {
    const full = join(dir, entry);
    const st = statSync(full);
    if (st.isDirectory()) {
      walkTsFiles(full, out);
    } else if (
      entry.endsWith('.ts') &&
      !entry.endsWith('.test.ts') &&
      !entry.endsWith('.spec.ts') &&
      !entry.endsWith('.d.ts')
    ) {
      out.push(full);
    }
  }
  return out;
}

interface MutationSite {
  file: string;
  line: number;
  col: number;
  method: string;
  enclosingFn: string;
}

function findMutationSites(absFile: string, projectRoot: string): MutationSite[] {
  const src = readFileSync(absFile, 'utf8');
  const sf = ts.createSourceFile(absFile, src, ts.ScriptTarget.ESNext, true);
  const rel = relative(projectRoot, absFile).split(sep).join('/');
  const sites: MutationSite[] = [];

  function nameOf(node: ts.Node): string | undefined {
    if ((ts.isFunctionDeclaration(node) || ts.isMethodDeclaration(node)) && node.name) {
      return node.name.getText(sf);
    }
    return undefined;
  }

  function visit(node: ts.Node, enclosing: string): void {
    let nextEnclosing = enclosing;
    const own = nameOf(node);
    if (own) nextEnclosing = own;

    if (ts.isCallExpression(node) && ts.isPropertyAccessExpression(node.expression)) {
      const method = node.expression.name.text;
      if (MUTATION_METHODS.has(method)) {
        const { line, character } = sf.getLineAndCharacterOfPosition(node.getStart(sf));
        sites.push({ file: rel, line: line + 1, col: character + 1, method, enclosingFn: nextEnclosing });
      }
    }
    ts.forEachChild(node, (c) => visit(c, nextEnclosing));
  }

  visit(sf, '<module>');
  return sites;
}

describe('scheduler event-queue boundary', () => {
  const projectRoot = process.cwd();
  const srcRoot = join(projectRoot, 'src');
  const files = walkTsFiles(srcRoot);
  const sites = files.flatMap((f) => findMutationSites(f, projectRoot));

  // Emit one it() per site so failures point at an exact file:line.
  for (const site of sites) {
    it(`${site.file}:${site.line} graph.${site.method}() is allowlisted`, () => {
      const fileAllow = (allowlist as Record<string, string[]>)[site.file] ?? [];
      const allowed = fileAllow.includes('*') || fileAllow.includes(site.enclosingFn);
      expect(
        allowed,
        `graph mutation at ${site.file}:${site.line}:${site.col} bypasses the event queue ŌĆö route through SchedulerLoop.enqueue() or add to test/integration/scheduler-boundary-allowlist.json`,
      ).toBe(true);
    });
  }

  it('emits at least one mutation site (sanity ŌĆö walker is working)', () => {
    expect(sites.length).toBeGreaterThan(0);
  });
});
