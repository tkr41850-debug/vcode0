const WORK_VALUES: WorkControl[] = ["discussing", "researching", "planning", "executing", "executing_repair", "ci_check", "verifying", "awaiting_merge", "summarizing", "work_complete"];
const COLLAB_VALUES: CollabControl[] = ["none", "branch_open", "merge_queued", "integrating", "merged", "conflict", "cancelled"];

// Expected-legal set derived from docs/architecture/data-model.md. Start with explicit allow-list; everything else is implicitly illegal.
const EXPECTED_LEGAL = new Set<string>([

describe("compositeGuard exhaustive matrix", () => {
  for (const work of WORK_VALUES) {
    for (const collab of COLLAB_VALUES) {
      for (const run of RUN_VALUES) {
