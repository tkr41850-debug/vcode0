New module (src/runtime/worktree/pid-registry.ts):

```typescript
import type { Store } from '@orchestrator/ports';
import type { AgentRunId } from '@core/types/ids';

export interface WorkerPidRegistry {
  set(agentRunId: AgentRunId, pid: number): void;
  clear(agentRunId: AgentRunId): void;
  list(): Array<{ agentRunId: AgentRunId; pid: number }>;
  isAlive(pid: number): boolean;
}

export function createWorkerPidRegistry(store: Store): WorkerPidRegistry {
  return {
    set: (id, pid) => store.setWorkerPid(id, pid),
    clear: (id) => store.clearWorkerPid(id),
    list: () => store.getLiveWorkerPids(),
    isAlive: (pid) => {
      try { process.kill(pid, 0); return true; }
      catch (err) {
        if ((err as NodeJS.ErrnoException).code === 'ESRCH') return false;
        if ((err as NodeJS.ErrnoException).code === 'EPERM') return true;  // exists but not ours
        throw err;
      }
    },
  };
}
```

```typescript
async removeWorktree(branch: string): Promise<void> {
  const worktreePath = this.pathFor(branch);  // reuse existing path helper
  try {
    await this.git.raw(['worktree', 'remove', '--force', worktreePath]);
  } catch (err) {
    const msg = (err as Error).message ?? '';
    // Idempotent: 'not a working tree' / 'No such file or directory' → success
    if (/not a working tree|no such file|does not exist/i.test(msg)) return;
    throw err;
  }
}

async pruneStaleWorktrees(): Promise<string[]> {
  // `git worktree prune -v` emits lines like "Removing worktrees/feat-x: ..."
  const raw = await this.git.raw(['worktree', 'prune', '-v']);
  const removed: string[] = [];
  for (const line of raw.split('\n')) {
    const m = /^Removing worktrees\/([^:]+):/.exec(line);
    if (m) removed.push(m[1]!);
  }
  return removed;
}

async sweepStaleLocks(isAlive: (pid: number) => boolean): Promise<string[]> {
  // Scan .git/worktrees/*/locked and .git/worktrees/*/gitdir
  // For each locked marker whose owning PID is dead, remove the lock.
  // PID is NOT always stored in the `locked` file by git; we fall back to
  // "remove lock if the referenced worktree dir no longer exists OR its
  // HEAD file is stale". Conservative: if unsure, leave the lock.
  const cleared: string[] = [];
  const worktreesDir = path.join(this.gitDir, 'worktrees');
  let entries: string[];
  try {
    entries = await fs.readdir(worktreesDir);
  } catch (err) {
    if ((err as NodeJS.ErrnoException).code === 'ENOENT') return cleared;
    throw err;
  }
  for (const name of entries) {
    const lockFile = path.join(worktreesDir, name, 'locked');
    const gitdirFile = path.join(worktreesDir, name, 'gitdir');
    try {
      await fs.access(lockFile);  // lock present
    } catch { continue; }
    // Check if the worktree's target gitdir still exists on disk.
    let targetAlive = true;
    try {
      const gitdir = (await fs.readFile(gitdirFile, 'utf-8')).trim();
      await fs.access(gitdir);
    } catch {
      targetAlive = false;
    }
    if (!targetAlive) {
      await fs.rm(lockFile, { force: true });
      cleared.push(name);
    }
    // Note: isAlive(pid) parameter is available for future refinement when
    // we stamp the PID into the lock file body. Current git does not do this
    // reliably; we rely on target-dir liveness instead. Parameter kept for
    // Phase 9 forward-compatibility.
  }
  return cleared;
}
```

- The `gitDir` path is usually `<repoRoot>/.git`; if the provisioner already has this (most likely constructor-injected), reuse it. Otherwise expose via the `simple-git` instance.
- The `isAlive` parameter is the caller's bridge to the PID registry — today used only for the "conservative when lock owner unknown" comment; Phase 9 may extend this to stamp and read PIDs. Keep the parameter shape stable.
