  return {
    messageCount: agent.state.messages.length,
    lastRole: agent.state.messages.at(-1)?.role,
    pendingToolCalls: agent.state.pendingToolCalls?.length ?? 0,
    isStreaming: agent.state.isStreaming,
    errorMessage: agent.state.errorMessage,

  it('Scenario 1 — cold start: prompt+submit completes with terminal tool_call', async () => {
    const agent = makeAgent(ALL_SCENARIOS.coldStart);
    await agent.prompt('task: do x');

    const agent = makeAgent(ALL_SCENARIOS.midToolCall);

    // Wait for tool_execution_start (but not end) then abort.
    await new Promise<void>((resolve) => {
      const unsub = agent.subscribe((ev) => {
        if (ev.type === 'tool_execution_start') {

          unsub();

import * as fs from 'node:fs/promises';
import * as path from 'node:path';

export interface ToolOutputStore {
  record(toolCallId: string, output: unknown): void | Promise<void>;
  get(toolCallId: string): unknown | undefined;
  clear(): void;
}

export function createInMemoryToolOutputStore(): ToolOutputStore {
  const map = new Map<string, unknown>();
  return {
    record(id, out) { map.set(id, out); },
    get(id) { return map.get(id); },
    clear() { map.clear(); },
  };
}

export function createFileToolOutputStore(dir: string): ToolOutputStore {
  return {
    async record(id, out) {
      await fs.mkdir(dir, { recursive: true });
      await fs.writeFile(path.join(dir, `${id}.json`), JSON.stringify(out));
    },
    get(id) {
      // Sync read for the splice path (called during resume, not hot path).
      try {
        const raw = require('node:fs').readFileSync(path.join(dir, `${id}.json`), 'utf-8');
        return JSON.parse(raw);
      } catch { return undefined; }
    },
    clear() {
      try { require('node:fs').rmSync(dir, { recursive: true, force: true }); } catch {}
    },
  };
}

import type { Agent, AgentMessage } from '@mariozechner/pi-agent-core';
import type { ToolOutputStore } from './tool-output-store.js';

export const RESUME_STRATEGY = 'persist-tool-outputs' as const;

export async function resume(opts: {
  agent: Agent;
  savedMessages: AgentMessage[];
  toolOutputs: ToolOutputStore;
}): Promise<void> {
  // Strategy: if the last message is an assistant message AND it referenced
  // tool calls whose results are missing from the transcript, splice them
  // from the tool-output store before calling continue().
  const messages = [...opts.savedMessages];
  const last = messages.at(-1);
  if (last?.role === 'assistant' && 'toolCalls' in last && last.toolCalls?.length) {
    for (const tc of last.toolCalls) {
      const saved = opts.toolOutputs.get(tc.id);
      if (saved !== undefined) {
        messages.push({ role: 'tool', toolCallId: tc.id, content: JSON.stringify(saved) } as any);
      }
    }
    // Re-init the agent's messages from the spliced set.
    (opts.agent.state as any).messages = messages;
  }
  await opts.agent.continue();
  await opts.agent.waitForIdle();
}
