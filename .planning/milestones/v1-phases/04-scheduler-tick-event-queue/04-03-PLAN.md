/**
 * Generate a synthetic graph of `featureCount` features × `tasksPerFeature` tasks.
 * Features are chained in groups of 5 (roughly matching typical project shape):
 *   group 1: f-0 → f-1 → f-2 → f-3 → f-4 (linear)
 *   group 2: f-5 parallel to group 1 tail
 * Half of features are pre-execution (virtual nodes); half are executing (task expansion).
 */
export function largeGraphFixture(opts: { featureCount: number; tasksPerFeature: number }): SchedulerFixture {
  const graph = new InMemoryFeatureGraph();
  withTick(graph, () => {
    graph.createMilestone({ id: 'm-bulk', name: 'Bulk', features: [] });
    for (let i = 0; i < opts.featureCount; i++) {
      const id = `f-${i}`;
      const depIds = i >= 5 && i % 5 !== 0 ? [`f-${i - 1}`] : [];
      graph.createFeature({ id, milestoneId: 'm-bulk', name: `F${i}`, dependsOn: depIds });
      // Half executing, half pre-execution
      if (i % 2 === 0) {
        graph.transitionFeature(id, { workControl: 'executing', collabControl: 'branch_open' });
        for (let t = 0; t < opts.tasksPerFeature; t++) {
          graph.addTask(id, { id: `t-${id}-${t}`, status: t === 0 ? 'ready' : 'blocked', dependsOn: t > 0 ? [`t-${id}-${t - 1}`] : [] });
        }
      } else {
        // Pre-execution feature — no task expansion; virtual node will aggregate weight.
        for (let t = 0; t < opts.tasksPerFeature; t++) {
          graph.addTask(id, { id: `t-${id}-${t}`, status: 'planned', dependsOn: t > 0 ? [`t-${id}-${t - 1}`] : [] });
        }
      }
    }
  });
  return { graph, expectedMetrics: new Map(), description: `largeGraph ${opts.featureCount}x${opts.tasksPerFeature}` };
}
