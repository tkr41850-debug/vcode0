```typescript
// Refactored signature (breaking at the function level, but prioritizeReadyWork already has the run reader):
function isRetryEligible(
  unit: SchedulableUnit,
  runReader: RunReader,
  config: { baseDelayMs: number; maxDelayMs: number; retryCap: number },
  now: number,
): boolean {
  const run = runReader.getExecutionRun(unit.kind === 'task' ? `run-task:${unit.task.id}` : `run-feature:${unit.feature.id}:${unit.phase}`);
  if (run === undefined) return false;
  if (run.runStatus !== 'retry_await') return false;
  const attempts = run.restartCount ?? 0;
  if (attempts >= config.retryCap) return false;
  const lastFailedAt = run.retryAt !== undefined ? (run.retryAt - computeBackoff(attempts, config)) : 0;
  // Simpler: if runStatus='retry_await' AND retryAt is set, eligibility is `now >= retryAt`.
  if (run.retryAt !== undefined && now < run.retryAt) return false;
  return true;
}

function computeBackoff(attempts: number, config: { baseDelayMs: number; maxDelayMs: number }): number {
  return Math.min(config.baseDelayMs * (2 ** attempts), config.maxDelayMs);
}
```
